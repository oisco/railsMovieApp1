TTITLE 'Countries and Locations'
SET PAGESIZE 40
SET LINESIZE 110
COLUMN location_id FORMAT 99999999 WORD_WRAPPED HEADING 'Location Id'
COLUMN city FORMAT A20 WORD_WRAPPED HEADING 'City'
COLUMN state_province FORMAT A20 WORD_WRAPPED HEADING 'Province'
COLUMN country_id FORMAT A10 WORD_WRAPPED HEADING 'Country Id'
COLUMN country_name FORMAT A40 WORD_WRAPPED HEADING 'Country'
SELECT LOCATIONS.location_id, LOCATIONS.city, LOCATIONS.state_province, 
LOCATIONS.country_id, COUNTRIES.country_name
FROM LOCATIONS JOIN COUNTRIES
ON LOCATIONS.country_id = COUNTRIES.country_id
ORDER BY LOCATIONS.location_id;
